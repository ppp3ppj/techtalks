                    newjeans.yml
                    ---
                    - name: Display Newjeans
                      hosts: localhost
                      tasks:
                        - name: Print members
                          debug:
                            msg: "{{ item }}"
                          loop:
                            - "Minji"
                            - "Haerin"
                            - "Hanni"
                            - "Danielle"
                            - "Hyein"
                    

                    newjeans.yml
                    ---
                    - name: Install Newjeans
                      hosts: localhost
                      become: yes # Run this with sudo
                      tasks:
                        - name: Install members pkg
                          apt: # Debian base
                            name: "{{ item }}"
                            state: latest
                            update_cache: yes
                          loop:
                            - "Minji"
                            - "Haerin"
                            - "Hanni"
                            - "Danielle"
                            - "Hyein"
