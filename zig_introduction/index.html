                                cat.mood = format!("Mood {} from thread {}", i, i);
                                println!("Thread {} updated cat: {:?}", i, *cat);
                                })
                                })
                                .collect();

                                for thread in threads {
                                thread.join().unwrap();
                                }

                                let final_cat = thread_safe_cat.lock().unwrap();
