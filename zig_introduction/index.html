Ownership and Concurrency in Rust


                            use std::cell::RefCell;
                            use std::rc::Rc;
                            use std::sync::{Arc, Mutex};
                            use std::thread;

                            #[derive(Debug)]
                            struct Cat {
                            name: String,
                            mood: String,
                            }

                            fn main() {
                            // 1. A single cat in a Box
                            let boxed_cat = Box::new(Cat {
                            name: "Whiskers".to_string(),
                            mood: "Happy".to_string(),
                            });
                            println!("Boxed cat: {:?}", boxed_cat);

                            // 2. Shared cat using Rc
                            let shared_cat = Rc::new(RefCell::new(Cat {
                            name: "Fluffy".to_string(),
                            mood: "Curious".to_string(),
                            }));
                            {
                            let cat = shared_cat.borrow();
                            println!("Shared cat: {:?}", cat);
                            }

                            // 3. Mutate the shared cat using RefCell
                            {
                            let mut cat = shared_cat.borrow_mut();
                            cat.mood = "Playful".to_string();
                            println!("Cat after mood change: {:?}", cat);
                            }

                            // 4. Use Mutex and Arc for thread-safe sharing
                            let thread_safe_cat = Arc::new(Mutex::new(Cat {
                            name: "Tiger".to_string(),
                            mood: "Lazy".to_string(),
                            }));

                            let threads: Vec<_> = (0..3)
                                .map(|i| {
                                let cat_clone = Arc::clone(&thread_safe_cat);
                                thread::spawn(move || {
                                let mut cat = cat_clone.lock().unwrap();
                                cat.mood = format!("Mood {} from thread {}", i, i);
                                println!("Thread {} updated cat: {:?}", i, *cat);
                                })
                                })
                                .collect();

                                for thread in threads {
                                thread.join().unwrap();
                                }

                                let final_cat = thread_safe_cat.lock().unwrap();
                                println!("Final cat state: {:?}", *final_cat);
                                }

Two Ways to Handle Mischief

1. Use try (Catch the Cat Gently üêæ)


                            pub fn main() anyerror!void {
                            const file = try std.fs.cwd().openFile("minji_haerin.txt", .{});
                            std.debug.print("Opened file: {}\n", .{file});
                            }

If the file doesn‚Äôt exist, Zig sends the error up the chain. You can handle it later.

2. Use catch (Tell the Cat to Chill üê±)


                            pub fn main() void {
                            const file = std.fs.cwd().openFile("minji_haerin.txt", .{}) catch |err| {
                            std.debug.print("Oops! Error: {}\n", .{err});
                            return;
                            };
                            std.debug.print("Opened file: {}\n", .{file});
                            }

If there‚Äôs an error, you catch it on the spot and deal with it.
